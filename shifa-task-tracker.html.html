// Shifa Medical Group — Task Tracker Apps Script
// Paste this in Google Apps Script (Extensions → Apps Script)
// Then Deploy → New Deployment → Web App → Execute as: Me → Anyone

const SPREADSHEET_ID = "YOUR_SPREADSHEET_ID_HERE";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const action = data.action;
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

    if (action === "login") return respond(handleLogin(ss, data));
    const user = verifyUser(ss, data.email, data.secret);
    if (!user) return respond({ok:false, message:"Authentication failed"});

    if (action === "getUsers")       return respond(getUsers(ss));
    if (action === "getTasks")       return respond(getTasks(ss, data, user));
    if (action === "getTaskById")    return respond(getTaskById(ss, data));
    if (action === "createTask")     return respond(createTask(ss, data, user));
    if (action === "updateTask")     return respond(updateTask(ss, data, user));
    if (action === "getTaskUpdates") return respond(getTaskUpdates(ss, data));
    if (action === "getActivity")    return respond(getActivity(ss, data));
    if (action === "getStats")       return respond(getStats(ss, user));

    return respond({ok:false, message:"Unknown action"});
  } catch(err) {
    return respond({ok:false, message:err.message});
  }
}

function respond(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function handleLogin(ss, data) {
  const sheet = ss.getSheetByName("Users");
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    const [email, name, secret, role, active] = rows[i];
    if (email.toLowerCase() === data.email.toLowerCase() &&
        secret === data.secret && active === true) {
      return {ok:true, user:{email, name, role}};
    }
  }
  return {ok:false, message:"Invalid credentials"};
}

function verifyUser(ss, email, secret) {
  const sheet = ss.getSheetByName("Users");
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0].toLowerCase() === email.toLowerCase() &&
        rows[i][2] === secret && rows[i][4] === true) {
      return {email:rows[i][0], name:rows[i][1], role:rows[i][3]};
    }
  }
  return null;
}

function getUsers(ss) {
  const sheet = ss.getSheetByName("Users");
  const rows = sheet.getDataRange().getValues();
  const users = rows.slice(1)
    .filter(r => r[4] === true)
    .map(r => ({email:r[0], name:r[1], role:r[3]}));
  return {ok:true, users};
}

function getTasks(ss, data, user) {
  const sheet = ss.getSheetByName("Tasks");
  const rows = sheet.getDataRange().getValues();
  let tasks = rows.slice(1).map(r => rowToTask(r));

  if (data.assigned_to) tasks = tasks.filter(t => t.assigned_to === data.assigned_to);
  if (data.status) tasks = tasks.filter(t => t.status === data.status);
  if (data.priority) tasks = tasks.filter(t => t.priority === data.priority);
  if (data.category) tasks = tasks.filter(t => t.category === data.category);
  if (data.mine) tasks = tasks.filter(t => t.assigned_to === user.email);

  tasks.sort((a,b) => {
    const priOrder = {Urgent:0, High:1, Medium:2, Low:3};
    return (priOrder[a.priority]||9) - (priOrder[b.priority]||9);
  });
  return {ok:true, tasks};
}

function getTaskById(ss, data) {
  const sheet = ss.getSheetByName("Tasks");
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(data.task_id)) {
      return {ok:true, task:rowToTask(rows[i])};
    }
  }
  return {ok:false, message:"Task not found"};
}

function rowToTask(r) {
  return {
    task_id:r[0], title:r[1], description:r[2], category:r[3], priority:r[4],
    status:r[5], progress_percent:Number(r[6]||0), created_by:r[7], assigned_to:r[8],
    created_at:r[9], due_date:r[10], last_updated_by:r[11], last_updated_at:r[12],
    latest_note:r[13], completed_at:r[14]
  };
}

function createTask(ss, data, user) {
  const sheet = ss.getSheetByName("Tasks");
  const taskId = "T" + Date.now();
  const now = new Date().toISOString();
  sheet.appendRow([
    taskId, data.title, data.description||"", data.category, data.priority||"Medium",
    "Not Started", 0, user.email, data.assigned_to, now, data.due_date||"",
    user.email, now, data.note||"", ""
  ]);
  if (data.note) {
    logUpdate(ss, taskId, user.email, "", "Not Started", 0, 0, data.note);
  }
  return {ok:true, task_id:taskId};
}

function updateTask(ss, data, user) {
  const sheet = ss.getSheetByName("Tasks");
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]) === String(data.task_id)) {
      const oldStatus = rows[i][5];
      const oldProg = rows[i][6];
      const newStatus = data.status || rows[i][5];
      const newProg = data.progress_percent !== undefined ? Number(data.progress_percent) : rows[i][6];
      const now = new Date().toISOString();
      const completedAt = newStatus === "Completed" ? (rows[i][14] || now) : "";

      sheet.getRange(i+1, 6).setValue(newStatus);
      sheet.getRange(i+1, 7).setValue(newProg);
      sheet.getRange(i+1, 10).setValue(data.due_date || rows[i][10]);
      sheet.getRange(i+1, 12).setValue(user.email);
      sheet.getRange(i+1, 13).setValue(now);
      sheet.getRange(i+1, 14).setValue(data.note || rows[i][13]);
      sheet.getRange(i+1, 15).setValue(completedAt);

      logUpdate(ss, data.task_id, user.email, oldStatus, newStatus, oldProg, newProg, data.note||"");
      return {ok:true};
    }
  }
  return {ok:false, message:"Task not found"};
}

function logUpdate(ss, taskId, updatedBy, oldStatus, newStatus, oldProg, newProg, note) {
  const sheet = ss.getSheetByName("Task_Updates");
  sheet.appendRow([
    "U"+Date.now(), taskId, updatedBy, new Date().toISOString(),
    oldStatus, newStatus, oldProg, newProg, note
  ]);
}

function getTaskUpdates(ss, data) {
  const sheet = ss.getSheetByName("Task_Updates");
  const rows = sheet.getDataRange().getValues();
  const updates = rows.slice(1)
    .filter(r => String(r[1]) === String(data.task_id))
    .map(r => ({
      update_id:r[0], task_id:r[1], updated_by:r[2], updated_at:r[3],
      old_status:r[4], new_status:r[5], old_progress:r[6], new_progress:r[7], note:r[8]
    }))
    .reverse();
  return {ok:true, updates};
}

function getActivity(ss, data) {
  const tasksSheet = ss.getSheetByName("Tasks");
  const updSheet = ss.getSheetByName("Task_Updates");
  const taskRows = tasksSheet.getDataRange().getValues();
  const taskMap = {};
  taskRows.slice(1).forEach(r => { taskMap[r[0]] = r[1]; });

  const rows = updSheet.getDataRange().getValues();
  const limit = data.limit || 50;
  const activity = rows.slice(1).map(r => ({
    update_id:r[0], task_id:r[1], task_title:taskMap[r[1]]||r[1],
    updated_by:r[2], updated_at:r[3],
    old_status:r[4], new_status:r[5], old_progress:r[6], new_progress:r[7], note:r[8]
  })).reverse().slice(0, limit);
  return {ok:true, activity};
}

function getStats(ss, user) {
  const sheet = ss.getSheetByName("Tasks");
  const rows = sheet.getDataRange().getValues().slice(1);
  const today = new Date(); today.setHours(0,0,0,0);
  let open=0, inProgress=0, overdue=0, urgent=0, done=0, dueToday=0;
  const byStatus={}, byCat={};

  rows.forEach(r => {
    const status = r[5], priority = r[4], due = r[10]?new Date(r[10]):null;
    if (due) due.setHours(0,0,0,0);
    const active = status !== "Completed" && status !== "Cancelled";

    if (active) open++;
    if (status === "In Progress") inProgress++;
    if (status === "Completed") done++;
    if (priority === "Urgent" && active) urgent++;
    if (due && due < today && active) overdue++;
    if (due && due.getTime() === today.getTime() && active) dueToday++;

    byStatus[status] = (byStatus[status]||0) + 1;
    if (r[3]) byCat[r[3]] = (byCat[r[3]]||0) + 1;
  });

  return {ok:true, stats:{open, inProgress, overdue, urgent, done, dueToday, byStatus, byCat}};
}
