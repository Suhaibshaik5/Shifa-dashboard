// Shifa Employee Tracker — Apps Script
const SPREADSHEET_ID = "YOUR_SPREADSHEET_ID";
function doPost(e){
  try{
    const d=JSON.parse(e.postData.contents),ss=SpreadsheetApp.openById(SPREADSHEET_ID);
    if(d.action==="login")return R(login(ss,d));
    const u=verify(ss,d.email,d.secret);if(!u)return R({ok:false,message:"Auth failed"});
    const m={getEmployees:()=>getEmps(ss,d),saveEmployee:()=>saveEmp2(ss,d,u),getDailyLogs:()=>getLogs(ss,d),saveDailyLog:()=>saveLog(ss,d,u),saveBulkLog:()=>saveBulk(ss,d,u),getWorkStatus:()=>getWS(ss),updateWorkStatus:()=>updateWS(ss,d,u),getPerformanceNotes:()=>getPerf(ss,d),savePerformanceNote:()=>savePerf2(ss,d,u),getStats:()=>stats(ss)};
    return R(m[d.action]?m[d.action]():{ok:false,message:"Unknown action"});
  }catch(err){return R({ok:false,message:err.message});}
}
function R(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);}
function login(ss,d){const rows=ss.getSheetByName("Users").getDataRange().getValues();for(let i=1;i({emp_id:r[0],email:r[1],name:r[2],role:r[3],department:r[4],phone:r[5],start_date:r[6],employment_type:r[7],emergency_contact:r[8],active:r[9]}));if(d.active!==undefined)e=e.filter(x=>x.active===d.active);return{ok:true,employees:e};}
function saveEmp2(ss,d,u){const sh=ss.getSheetByName("Employees"),rows=sh.getDataRange().getValues();for(let i=1;i({log_id:r[0],employee_email:r[1],date:r[2],status:r[3],check_in:r[4],check_out:r[5],hours_worked:r[6],notes:r[7],logged_by:r[8],logged_at:r[9]}));if(d.date)l=l.filter(x=>String(x.date)===d.date);if(d.employee_email)l=l.filter(x=>x.employee_email===d.employee_email);if(d.month&&d.year)l=l.filter(x=>String(x.date).startsWith(d.year+"-"+d.month.padStart(2,"0")));return{ok:true,logs:l};}
function saveLog(ss,d,u){const sh=ss.getSheetByName("Daily_Logs"),rows=sh.getDataRange().getValues();for(let i=1;isaveLog(ss,e,u));return{ok:true};}
function getWS(ss){const rows=ss.getSheetByName("Work_Status").getDataRange().getValues().slice(1);return{ok:true,statuses:rows.map(r=>({employee_email:r[0],work_status:r[1],current_task:r[2],note:r[3],updated_at:r[4]}))}}
function updateWS(ss,d,u){const sh=ss.getSheetByName("Work_Status"),rows=sh.getDataRange().getValues();for(let i=1;i({note_id:r[0],employee_email:r[1],date:r[2],rating:r[3],category:r[4],type:r[5],note:r[6],logged_by:r[7],logged_at:r[8]}));if(d.employee_email)n=n.filter(x=>x.employee_email===d.employee_email);if(d.type)n=n.filter(x=>x.type===d.type);return{ok:true,notes:n.reverse()};}
function savePerf2(ss,d,u){ss.getSheetByName("Performance_Notes").appendRow(["N"+Date.now(),d.employee_email,d.date,d.rating,d.category,d.type,d.note,u.email,new Date().toISOString()]);return{ok:true};}
function stats(ss){const today=new Date().toISOString().slice(0,10);const logs=ss.getSheetByName("Daily_Logs").getDataRange().getValues().slice(1).filter(r=>String(r[2])===today);const emps=ss.getSheetByName("Employees").getDataRange().getValues().slice(1).filter(r=>r[9]===true);const c={Present:0,Late:0,Absent:0,Remote:0,"Half Day":0,"On Leave":0};logs.forEach(r=>{if(c[r[3]]!==undefined)c[r[3]]++;});return{ok:true,stats:{...c,total:emps.length}};}
